package 数据结构;

import java.util.Arrays;
import java.util.Scanner;

public class HashTab {
    public static void main(String[] args) {
        HashT tab=new HashT(5);
        Emp p1=new Emp(1,"李华");
        Emp p2=new Emp(2,"张三");
        Scanner c=new Scanner(System.in);
        tab.add(p1);
        tab.add(p2);
        tab.show();
        tab.FindEmp(2);
    }
}
//开始创建哈希表
class HashT{
    private empLinkedList[] empLinkedArr;
    private int size;
    public HashT(int size)
    {
        this.size=size;
        empLinkedArr=new empLinkedList[size];
        //分别初始化每个链表
        for (int i = 0; i < size; i++) {
            empLinkedArr[i]=new empLinkedList();
        }
    }
    public void add(Emp p)
    {
        //根据id，判断添加到哪
        int EmNo=hashFun(p.id);
        empLinkedArr[EmNo].add(p);
    }
    public int hashFun(int id)
    {
        return id%size;
    }
    public void show()
    {
        for(int i=0;i<size;i++)
        {
            empLinkedArr[i].show(i);
        }
    }
    //查找，如果找到，返回emp，没有找到返回null
    public void FindEmp(int id)
    {
        int no=hashFun(id);
        Emp p=empLinkedArr[no].find(id);
        if(p!=null)
        {
            System.out.println("找到");
        }
        else{
            System.out.println("未找到");
        }
    }
}
class Emp{
    public int id;
    public String name;
    public Emp next;
    public Emp(int id,String name)
    {
        super();
        this.id=id;
        this.name=name;
    }
}
class empLinkedList{
    //头指针指向第一个emp
    private Emp head;

    public void add(Emp p)
    {
        if(head==null)
        {
            head=p;
            System.out.println("添加成功！");
            return;
        }
        Emp cur=head.next;
        while(true)
        {
            if(cur==null)
            {
                break;
            }
            cur=cur.next;
        }
        cur=p;
        System.out.println("添加成功！");
    }
    public void show(int no)
    {
        if(head==null)
        {
            System.out.println(no+"条链表为空");
        }
        Emp temp=head;
        while(true)
        {
            if(temp==null)
            {
                break;
            }
            System.out.print("id:"+temp.id+"   "+"name:"+temp.name+"  ");
            System.out.println();
            temp=temp.next;
        }
    }
    public Emp find(int id)
    {
        if(head==null)
            return null;
        Emp temp=head;
        while(true)
        {
            if(temp==null)
                break;
            if(temp.id==id)
            {
                return temp;
            }
            temp=temp.next;
        }
        return null;
    }
}
