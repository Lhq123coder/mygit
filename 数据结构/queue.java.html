public class queue {
    int max=8;
    int[] arr=new int[max];
    public static void main(String[] args) {
        queue q=new queue();
        q.check(0);

    }
    //放置皇后
    private void check(int n)
    {
        if(n==max)
        {
            print();
            return;
        }
        //依次放入并判断是否冲突
        for (int i = 0; i < max; i++) {
            arr[n]=i;
            if(judge(n))
            {
               check(n+1);
            }
        }
    }
    //查看是否冲突
    private boolean judge(int n)
    {
        for(int i=0;i<n;i++)
        {
            if(arr[i]==arr[n]||Math.abs(n-i)==Math.abs(arr[n]-arr[i]))
            {
                return false;
            }
        }
        return true;
    }
    public void print()
    {
        for (int i = 0; i < arr.length; i++) {
            System.out.print(arr[i]+" ");
        }
        System.out.println();
    }
}
