class doublelinkednode{
    public int size;
    private mynode head=new mynode();
    public void show()//显示
    {
        if(head.next==null)
        {
            System.out.println("链表为空");
            return;
        }
        mynode temp=head.next;
        while(true)
        {
            if(temp==null)
            {
                break;
            }
            System.out.print(temp.no+temp.name+"  ");
            temp=temp.next;
        }
        System.out.println();
    }
    public void add(mynode node)//添加
    {
        mynode temp=head;
        while(true)
        {
            if(temp.next==null)
            {
                break;
            }
            temp=temp.next;
        }
        temp.next=node;
        node.pre=temp;
    }
    public void change(mynode node)//修改
    {
        mynode temp=head.next;
        while(true)
        {
            if(temp==null)
            {
                System.out.println("未找到");
                break;
            }
            if(temp.no==node.no)
            {
                temp.name=node.name;
                break;
            }
            temp=temp.next;
        }
    }
    public void change(mynode node,int dex)//修改
    {
    }
    public void deletenode(mynode node)//删除
    {
        mynode temp=head.next;
        while(true)
        {
            if(node==null)
            {
                break;
            }
            if(node.no==temp.no)
            {
                temp.pre.next=node.next;
                node.next.pre=node.pre;
                break;
            }

            temp=temp.next;
        }

    }
}
class mynode{
    public int no;
    public String name;
    public mynode pre;
    public mynode next;
    public mynode(){

    }
    public mynode(int n,String name)
    {
        no=n;
        this.name=name;
        pre=null;
        next=null;
    }
}
