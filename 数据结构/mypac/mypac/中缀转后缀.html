package mypac;

import java.util.ArrayList;
import java.util.List;

//1.初始化两个栈，一个是运算符栈，一个是中间结果栈
//        2.从左至右遍历中缀表达式
//        3.遇到数字，压入栈二
//        4.遇到运算符：比较与s1栈顶运算符优先级
//        如果s1为空，货站定的元素为“（”，将该符号入栈
//        如果优先级比栈顶元素高，入栈
//        否则将s1栈顶运算符弹出到s2，再进行4.1进行比较
//        5.遇到括号
//        1》.若是左括号，直接压入栈
//        2》.若是右括号，将s1的栈顶弹到s2，直到有左括号，终止
//        6.重复进行2-5
//        7.将s1中的元素压入s2
//        8.将s2的元素返回
class hello{
    //将中缀表达式转为list
    public static List<String> toinfixexp(String s)
    {
        List<String> ls=new ArrayList<String>();
        int i=0;
        String str;//对多位数拼接
        char c;
        do{
            if((c=s.charAt(i))<48||(c=s.charAt(i))>57)
            {
                ls.add(""+c);
                i++;
            }
            else{
                str="";
                while(i<s.length()&&(c=s.charAt(i))>=48&&(c=s.charAt(i))<=57)
                {
                    str+=c;
                    i++;
                }
                ls.add(str);
            }

        }while(i<s.length());
        return ls;
    }
    public static void main(String[] args){
        String exp="1+2-4*(3+1)";
        List<String> in=toinfixexp(exp);
        System.out.println(in);
    }
}
