//环形链表,约瑟夫问题
/* 
1.先创建一个节点，让first指向该节点，形成环形 
2.后面每当我们创建一个新节点，就把该节点加入到已有的环形链表中即可。 
遍历： 
1.让辅助变量指向first 
2.while遍历该链表 
3.node.next==first,结束 
 
 */
class circlenodelist
{
    mynode first=null;
    public void add(int nums)
    {
        if(nums<1)
        {
            System.out.println("数据不对");
            return;
        }
        mynode cur=null;
        for (int i = 1; i <= nums; i++) {
            //创建小孩节点
            mynode b=new mynode(i);
            if(i==1)
            {
                first=b;
                first.next=first;
                cur=first;
            }
            else {
                cur.next=b;
                cur=b;
                cur.next=first;
            }
        }
    }
    public void show()
    {
        mynode temp=first;
        while(temp.next!=first)
        {
            System.out.print(temp.no+"   ");
            temp=temp.next;
        }
        System.out.print(temp.no+"   ");
    }
//出圈
    public void countnode(int start,int count,int nums)
    {
        if(first==null||start<1||start>nums)
        {
            System.out.println("参数输入有误，请重新输入");
            return;
        }
        mynode helper=first;
        while(true)
        {
            if(helper.next==first)
            {
                break;
            }
            helper=helper.next;
        }
        for (int i = 0; i < start-1; i++) {
            first=first.next;
            helper=helper.next;
        }
        while(true)
        {
            if(first==helper)
                break;
            //让first和helper移动count-1次，出圈
            for (int i = 0; i < count-1; i++) {
                first=first.next;
                helper=helper.next;
            }
            System.out.println("over"+first.no+"   ");
            first=first.next;
            helper.next=first;
        }
        System.out.println("end"+first.no);
    }
}
class mynode{
    public int no;
    public mynode next;
    public mynode(){

    }
    public mynode(int n)
    {
        no=n;
        next=null;
    }
}
