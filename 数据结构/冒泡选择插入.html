    void sort(int ...x) {//冒泡排序
        for (int i = 0; i < x.length - 1; i++) {
            for (int j = 0; j < x.length - i - 1; j++) {
                if (x[j] > x[j + 1]) {
                    int temp = x[j];
                    x[j] = x[j + 1];
                    x[j + 1] = temp;
                }
            }
        }
    }
    //优化
    void sort2(int ...x)//冒泡排序优化
    {
        boolean flag=false;
        for (int i = 0; i < x.length - 1; i++) {
            for (int j = 0; j < x.length - i - 1; j++) {
                if (x[j] > x[j + 1]) {
                    int temp = x[j];
                    x[j] = x[j + 1];
                    x[j + 1] = temp;
                    flag=true;
                }
            }
            if(flag==false)//在一趟排序中，如果一次都没排序，flag为false，后面不用循环
            {
                break;
            }
            else{
                flag=false;
            }
        }
    }
    void selectsort(int ...x)//时间复杂度低于冒泡
    {

        for (int i = 0; i < x.length-1; i++) {//遍历后面最小的，将它与第一个元素互换
            int min=x[i];
            int mindex=i;
            for (int j = i; j < x.length; j++) {
                if(x[j]<min)
                {
                    min=x[j];
                    mindex=j;
                }
            }
            x[mindex]=x[i];
            x[i]=min;
        }
    }
    void insertsort(int ...x){
        //插入排序
        //将一个序列看成一个有序和一个无序，从无序中拿元素插入有序表
        //逐步推导,第一轮
        for(int i=1;i<x.length;i++) {
            int insertvalue = x[i];
            int insertdex = i - 1;//和你前一个数进行比较
            while (insertdex >= 0&&insertvalue < x[insertdex]  ) {
                x[insertdex + 1] = x[insertdex];
                insertdex--;
            }
            x[insertdex + 1] = insertvalue;
        }

    }
}
